$$
\begin{aligned}
\frac{\partial \boldsymbol{p}{r}^{T}}{\partial z{t}} &= \frac{\partial \boldsymbol{p}{t}^{CT}}{\partial z{t}} \cdot \frac{\partial \boldsymbol{p}{r}^{CT}}{\partial \boldsymbol{p}{t}} \cdot \frac{\partial \boldsymbol{p}{r}^{T}}{\partial \boldsymbol{p}{r}^{C}}
\end{aligned}
$$

$$
\begin{aligned}
\begin{bmatrix}\boldsymbol{p}{r}\1\end{bmatrix} &= \frac{1}{z{r}} K \boldsymbol{p}{r}^{C} \
u{r} &= f_{x} \frac{x_{r}}{z_{r}} + u_{0} \
v_{r} &= f_{y} \frac{y_{r}}{z_{r}} + v_{0} \
\frac{\partial \boldsymbol{p}{r}^{T}}{\partial \boldsymbol{p}{r}^{C}} &= \begin{bmatrix}
\frac{\partial u_{r}}{\partial x_{r}} & \frac{\partial v_{r}}{\partial x_{r}} \
\frac{\partial u_{r}}{\partial y_{r}} & \frac{\partial v_{r}}{\partial y_{r}} \
\frac{\partial u_{r}}{\partial z_{r}} & \frac{\partial v_{r}}{\partial z_{r}} \
\end{bmatrix} \
&= \begin{bmatrix}
\frac{f_{x}}{z_{r}} & 0 \
0 & \frac{f_{y}}{z_{r}} \

  • \frac{f_{x}x_{r}}{z_{r}^{2}} & - \frac{f_{y}y_{r}}{z_{r}^{2}}
    \end{bmatrix} \
    &= \frac{1}{z_{r}} \begin{bmatrix}
    1 & 0 \
    0 & 1 \
  • \frac{x_{r}}{z_{r}} & - \frac{y_{r}}{z_{r}}
    \end{bmatrix} \boldsymbol{K}_{2}
    \end{aligned}
    $$

$$
\begin{aligned}
\boldsymbol{p}{r}^{C} &= \boldsymbol{R} \boldsymbol{p}{t}^{C} + \boldsymbol{t} \
\frac{\partial \boldsymbol{p}{r}^{CT}}{\partial \boldsymbol{p}{t}^{C}} &= \boldsymbol{R}^{T}
\end{aligned}
$$

$$
\begin{aligned}
\boldsymbol{p}{t}^{C} &= z{t} \boldsymbol{K}^{-1} \begin{bmatrix}\boldsymbol{p}{t} \ 1\end{bmatrix} \
\frac{\partial \boldsymbol{p}{t}^{CT}}{\partial z_{t}} &= \left( \boldsymbol{K}^{-1} \begin{bmatrix} \boldsymbol{p}{t} \ 1\end{bmatrix} \right)^{T} \
&= \frac{1}{z{t}} \boldsymbol{p}_{t}^{CT}
\end{aligned}
$$

$$
\begin{aligned}
\frac{\partial \boldsymbol{p}{r}^{T}}{\partial z{t}} &= \frac{1}{z_{t}} \boldsymbol{p}{t}^{CT} \boldsymbol{R}^{T} \cdot \frac{1}{z{r}} \begin{bmatrix}
1 & 0 \
0 & 1 \

  • \frac{x_{r}}{z_{r}} & - \frac{y_{r}}{z_{r}}
    \end{bmatrix} \boldsymbol{K}{2} \
    &= \frac{1}{z    {t}z_{r}} (\boldsymbol{p}_{r}^{C} - \boldsymbol{t})^{T} \begin{bmatrix}
    1 & 0 \
    0 & 1 \
  • \frac{x_{r}}{z_{r}} & - \frac{y_{r}}{z_{r}}
    \end{bmatrix} \boldsymbol{K}{2} \
    &= \frac{1}{z    {t}z_{r}} \begin{bmatrix} x_{r} & y_{r} & z_{r} - t \end{bmatrix} \begin{bmatrix}
    1 & 0 \
    0 & 1 \
  • \frac{x_{r}}{z_{r}} & - \frac{y_{r}}{z_{r}}
    \end{bmatrix} \boldsymbol{K}{2} \
    &= \frac{t}{z    {t}z_{r}^{2}} \begin{bmatrix} f_{x} x_{r} & f_{y} y_{r} \end{bmatrix} \
    &= \frac{t}{z_{t}z_{r}^{2}} \begin{bmatrix} u_{r} - u_{0} & v_{r} - v_{0} \end{bmatrix}
    \end{aligned}
    $$

$$
\frac{\partial L_{p}}{\partial z_{t}} = \frac{\partial \boldsymbol{p}{r}^{T}}{\partial z{t}} \frac{\partial L_{p}}{\partial \boldsymbol{p}_{r}}
$$
